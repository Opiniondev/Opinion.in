<script>
  function postOpinion() {
    const input = document.getElementById("opinionInput");
    const text = input.value.trim();
    if (text === "") return;

    const feed = document.getElementById("opinionFeed");

    const post = document.createElement("div");
    post.className = "opinion";
    post.innerHTML = `
      <p>${text}</p>
      <div class="opinion-actions">
        <button onclick="like(this)">üëç Like</button>
        <button onclick="dislike(this)">üëé Dislike</button>
      </div>
    `;

    feed.prepend(post);
    input.value = "";
  }

  function like(btn) {
    btn.innerHTML = `üëç Liked`;
  }

  function dislike(btn) {
    btn.innerHTML = `üëé Disliked`;
  }
